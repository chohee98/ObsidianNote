
### 가위바위보 게임
```cpp
#include <iostream>
using namespace std;

int main()
{
	const int SCISSORS = 1;
	const int ROCK = 2;
	const int PAPER = 3;
	
	string str[] = { "가위", "바위", "보" };
	int winRate = 0, win = 0, loose = 0;

	while (1)
	{
		cout << "가위(1) 바위(2) 보(3) 골라주세요!" << endl;
		cout << "> ";
		
		int randNum = rand() % 3 + 1;
		int chooseNum = 0;
		cin >> chooseNum;

		if (chooseNum - 1 > 3)
			break;
	
		cout << *(str+(chooseNum-1)) << "(님) vs " << *(str+(randNum-1)) << "(컴퓨터) ";
		if (chooseNum == randNum)
			cout << "비겼습니다";
		else if (chooseNum - randNum == -1 || chooseNum - randNum == 2)
		{
			cout << "졌습니다";
			loose++;
		}		
		else
		{
			cout << "이겼습니다";
			win++;
		}
		if (win + loose == 0)
			cout << "(현재 승률: 없음)" << endl;
		else
		{
			winRate = (win * 100) / (win + loose);
			cout << "(현재 승률: " << winRate << "%)" << endl;
		}	
	}
}
```
